#version 460

#include "BuiltIn.h"

layout (location = 0) in vec3 inPosition;
layout (location = 1) in vec4 inColor;

out gl_PerVertex
{
	vec4 gl_Position;
};
layout (location = 0) out vec4 outColor;

void main()
{
    gl_Position = ProjectionMatrix * ModelViewMatrix * vec4(inPosition, 1.0);	
    outColor = inColor;
    return;
}
